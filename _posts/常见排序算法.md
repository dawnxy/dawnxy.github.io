# 冒泡排序
**冒泡排序思想**：两两比较
每一项跟后一项进行比较，如果当前项大于后一项，交换位置，否则继续下一次比较

```js
var arr=[4,2,6,4,3,7,1];
function bubbleSort(arr) {
    for(var i=0;i<arr.length;i++){//比较的轮数
        for(var j=0;j<arr.length-1-i;j++){//每一轮最多比较的次数

                arr[j]=temp;*/
                arr[j]=arr[j]+arr[j+1];
                arr[j+1]=arr[j]-arr[j+1];
                arr[j]=arr[j]-arr[j+1];
                flag=true;//只要有交换位置，让flag=true
            }
        }
        if(flag){//还有交换位置的
            flag=false;//每一轮开始前，让flag=false;
        }else{//没有交换位置的了
            break;//当前循环结束
        }
    }
    return arr;
}
```

# 快速排序
**快速排序思想**：跟基准点比较
找到数组的基准点（一般将中间项作为基准点，向下取整（数组长度/2））
将数组中的每一项跟基准点进行比较，如果比基准点小，将该项放到一个空数组，如果比基准点大，将该项放到另一个空数组
直到数组中只有一个数为止

```js
var ary=[3,1,5,2,4,6,4,5,1,8,6,7];
function quickSort(arr){
    if(arr.length<=1){
        return arr;
    }
    var eg=Math.floor(arr.length/2);
    var val=arr.splice(eg,1)[0];
    var lAry=[],rAry=[];
    for(var i=0;i<arr.length;i++){
        if(arr[i]<val){
            lAry.push(arr[i]);
        }else{
            rAry.push(arr[i]);
        }
    }
    return quickSort(lAry).concat([val],quickSort(rAry));
}
var res=quickSort(ary);
console.log(res);
```

# 插入排序
**插入排序思想**：拿牌
拿到一张牌，放到左手
之后拿到的牌依次跟左手中的牌从后往前进行比较
如果小于当前牌，依次往前比...直到碰到比当前牌大的，放到当前牌之后

```js
var ary=[3,1,5,2,4,6,4,5,1,8,6,7];
function insertSort(arr){
    var newAry=[];
    newAry.push(arr[0]);//先抓第一张牌
    //依次把桌面上第二张及以后的牌抓到
    for(var i=1;i<arr.length;i++){
        var cur=arr[i];
        //抓到当前牌之后，分别从右往左跟左手中的牌进行比较
        for(var j=newAry.length-1;j>=0;){
            if(cur<newAry[j]){//当前新抓的牌比左手中的牌小
                j--;//继续和newAry[j]前面的牌比较
                if(j===-1){//说明当前拿的牌比左手中的牌都小
                    newAry.unshift(cur);//将这张牌放到开头位置
                }
            }else{//当前抓的牌比左手中newAry[j]牌大
                newAry.splice(j+1,0,cur);//将这张牌放在newAry[j]牌之后
                break;//不再进行比较，循环结束，此处也可以用j=-1,结束循环
            }
        }
    }
    return newAry;
}
var res=insertSort(ary);
console.log(res);
```

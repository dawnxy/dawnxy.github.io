---
title: 用多种方式实现求数组的最大值或最小值
date: 2017-03-17 18:44:20
categories: JavaScript
---

废话不多说，直接上源码
```js
var arr=[2,4,3,12,1,6,7,4,5];
//方法一：apply
var max=Math.max.apply(null,arr);

//方法二：spread运算符
var max=Math.max(...arr);

//方法三：reduce方法
var max=arr.reduce(function (a,b) {
    return Math.max(a,b);
});

//方法四：假设法
var max=arr[0];
for(var i=0;i<arr.length;i++){
    max=arr[i]>max?arr[i]:max;
}

//方法五：sort方法
var res=arr.sort(function (a,b) {
    return a-b;
});
var max=res[res.length-1];

//方法六：eval+Math.max+join
var max=eval('Math.max('+arr.join()+')');

console.log(max);
```
